integrate :: Formula -> Formula -> Formula Source #

integrate function

>>> let x = V "x"
>>> integrate x x
(1/2)*(x^2)
>>> integrate (x^2) x
(1/3)*(x^3)
>>> integrate (sin x) x
(-1)*(cos(x))

terms :: Formula -> Formula -> [Formula] Source #

get terms of formula

>>> let [x,y] = map V ["x","y"]
>>> terms x x
[x]
>>> terms y x
[]
>>> terms (sin x) x
[x,sin(x),cos(x)]
>>> terms (1/(sin x)) x
[x,sin(x),cos(x)]
